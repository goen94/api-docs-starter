# Deposit Cashback
## Deposit Cashback model

Deposit cashback model contains all the information about deposit cashback.

### Properties

<Row>
    <Col>
        <Properties>
            <Property name="_id" type="objectId">
                Unique identifier.
            </Property>
            <Property name="deposit_id" type="objectId">
                Identifier from deposit placement.
                - Is Required
            </Property>
            <Property name="depositCashback_id" type="objectId">
                Identifier from deposit cashbacks.
                - Is Required
            </Property>
            <Property name="formNumber" type="string">
                Cashback form number.
                - Auto Generated
            </Property>
            <Property name="receival" type="array">
                Cashback receivals.
            </Property>
            <Property name="receival.$._id" type="objectId">
                Unique identifier for cashback receival.
            </Property>
            <Property name="receival.$.date" type="timestamp">
                Cashback receival date.
                - Is Required
            </Property>
            <Property name="receival.$.received" type="number">
                Cashback received amount.
                - Is Required
            </Property>
            <Property name="receival.$.correctionAmount" type="number">
                Cashback correction amount.
            </Property>
            <Property name="receival.$.correctionReceived" type="number">
                Cashback correction received.
            </Property>
            <Property name="receival.$.remaining" type="number">
                Remaining cashback.
            </Property>
            <Property name="receival.$.createdBy_id" type="objectId">
                User identifier who create the deposit placement.
            </Property>
            <Property name="receival.$.createdAt" type="timestamp">
                Timestamp of when the deposit was created.
            </Property>
            <Property name="note" type="string">
                Cashback note.
            </Property>
            <Property name="createdBy_id" type="objectId">
                User identifier who create the deposit placement.
            </Property>
            <Property name="createdAt" type="timestamp">
                Timestamp of when the deposit was created.
            </Property>
            <Property name="updatedBy_id" type="objectId">
                User identifier who update the deposit placement.
            </Property>
            <Property name="updatedAt" type="timestamp">
                Timestamp of when the deposit was updated.
            </Property>
            <Property name="deletedBy_id" type="objectId">
                User identifier who deleted the deposit placement.
            </Property>
            <Property name="deletedAt" type="timestamp">
                Timestamp of when the deposit was deleted.
            </Property>
            <Property name="deletedReason" type="string">
                Deposit delete reason.
